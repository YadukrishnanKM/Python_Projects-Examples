Sin Tan relation¶

In [ ]:
import numpy as np
import matplotlib . pyplot as plt 
ax = plt.figure().add_subplot(projection='3d')
x = np.linspace(0, 1, 100)
y = np.sin(x * 2 * np.pi) / 2 + 0.5
ax.plot(x, y, zs=0, zdir='z', label='Sin in (x, y)')

x = np.linspace(0, 1, 100)
y = np.tan(x * 2 * np.pi) / 2 + 0.5
ax.plot(x, y, zs=0, zdir='y', label='Tan in (x, z)')
plt.legend()
plt.show()
